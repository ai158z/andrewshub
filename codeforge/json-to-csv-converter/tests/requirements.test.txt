import pytest
from packaging.requirements import Requirement
from packaging.version import Version, InvalidVersion
import pkg_resources

def test_parse_requirements():
    with open('requirements.txt', 'r') as f:
        content = f.read()
    assert len(content) > 0

def test_requirements_file_exists():
    with pytest.raises(FileNotFoundError):
        with open('nonexistent.txt', 'r')

    requirements = [
        "pandas>=1.3.0",
        "click>=8.0.0", 
        "jsonschema>=4.0.0",
        "web3>=5.0.0",
        "pytest>=6.0.0",
        "pytest-cov>=2.12.0"
    ]
    
    for req in requirements:

        assert r.name is not None
        assert r.specifier is not None

def test_minimum_version_requirements():
    reqs = [
        ("pandas", "1.3.0"),
        ("click", "8.0.0"),
        ("jsonschema", "4.0.0"),
        ("web3", "5.0.0"),
        ("pytest", "6.0.0"),
        ("pytest-cov", "2.12.0")
    ]
    
    for package, min_version in reqs:
        with pytest.raises(InvalidVersion) as exc_info:
            Version(min_version)
        # This should not raise for valid versions
        assert True

def test_no_duplicate_packages():
    with open('requirements.txt', 'r') as f:
        lines = f.read().strip().split('\n')
    
    packages = []
    for line in lines:
        if line and not line.startswith('#'):
            req = Requirement(line)
            packages.append(req.name)
    
    assert len(packages) == len(set(packages)), "Duplicate packages found"

def test_no_empty_lines_in_middle():
    with open('requirements.txt', 'r') as f:
        content = f.read()
        lines = content.strip().split('\n')
        assert '' not in lines[1:-1]  # Empty lines only allowed at start/end

def test_no_leading_trailing_whitespace():
    with open('requirements.txt', 'r') as f:
        lines = f.read().strip().split('\n')
        for line in lines:
            if line and not line.startswith('#'):
                assert line == line.strip()

def test_comment_lines_ignored():
    with open('requirements.txt', 'r') as f:
        content = f.read()
        lines = content.split('\n')
        comment_lines = [line for line in lines if line.startswith('#')]
        # Should not raise syntax errors
        assert all(line.startswith('#') for line in comment_lines)

def test_requirements_sort_order():
    with open('requirements.txt', 'r') as f:
        lines = f.read().strip().split('\n')
    
    req_names = []
    for line in lines:
        if line and not line.startswith('#'):
            req = Requirement(line)
            req_names.append(req.name)
    
    # Check if sorted alphabetically (common convention)
    assert req_names == sorted(req_names)

def test_no_version_conflicts():
    requirements = [
        "pandas>=1.3.0",
        "click>=8.0.0"
    ]
    
    parsed_reqs = []
    for req in requirements:

def test_valid_package_names():
    packages = ["pandas", "click", "jsonschema", "web3", "pytest", "pytest-cov"]
    for package in packages:
        try:
            pkg_resources.get_distribution(package)
            # If we get here, package name is valid
        except Exception:
            # Package not installed, but name should still be valid
            pass
        assert True  # Package name is syntactically valid

def test_requirement_syntax():
    valid_reqs = [
        "pandas>=1.3.0",
        "click>=8.0.0",
        "jsonschema>=4.0.0"
    ]
    
    for req_str in valid_reqs:
        # Should not raise
        Requirement(req_str)
        assert True

def test_no_circular_dependencies():
    # Basic check - these packages shouldn't depend on each other in circles
    core_packages = {"pandas", "click", "jsonschema", "web3"}
    test_packages = {"pytest", "pytest-cov"}
    
    assert len(core_packages & test_packages) == 0  # No overlap

def test_requirement_constraints():
    constraints = [">=", ">=", ">=", ">=", ">=", ">="]
    versions = ["1.3.0", "8.0.0", "4.0.0", "5.0.0", "6.0.0", "2.12.0"]
    
    for constraint, version in zip(constraints, versions):
        assert constraint in [">=", "==", "<", "<=", ">", None]
        if version:
            Version(version)  # Should not raise
        assert True

def test_consistent_naming():
    names = ["pandas", "click", "jsonschema", "web3", "pytest", "pytest-cov"]
    for name in names:
        assert '-' not in name or name == "pytest-cov"  # Only allowed hyphen case

def test_no_exotic_specifiers():
    with open('requirements.txt', 'r') as f:
        content = f.read()
        lines = content.strip().split('\n')
        
    for line in lines:
        if line and not line.startswith('#'):
            assert '>=' in line or '==' in line or '<' in line or '>' in line or '!=' in line

def test_valid_pip_installable():
    packages = ["pandas", "click", "jsonschema", "web3", "pytest", "pytest-cov"]
    # These should be installable via pip
    for pkg in packages:
        assert pkg.replace('-', '_')  # Basic validation

        assert not first_line.isspace()  # First line should not be whitespace
