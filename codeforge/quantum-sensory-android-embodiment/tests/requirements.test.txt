def test_no_duplicate_packages():
    """Test that there are no duplicate package entries"""
    with open('requirements.txt', 'r') as f:
        lines = [line.strip() for line in f.readlines() if line.strip() and not line.startswith('#')]
    
    # Check for exact duplicates
    assert len(lines) == len(set(lines)), "Duplicate package entries found"

def test_valid_package_format():
    """Test that all requirements are in valid format"""
    with open('requirements.txt', 'r') as f:
        lines = f.readlines()
    
    for line in lines:
        line = line.strip()
        if line and not line.startswith('#'):
            # Should be in format "package>=version" or "package==version" or just "package"
            assert '==' in line or '>=' in line or '<=' in line or line.count('=') == 0, f"Invalid format: {line}"

def test_no_empty_lines_in_middle():
    """Test that there are no empty lines in the middle of the file"""
    with open('requirements.txt', 'r') as f:
        content = f.read()
    
    lines = content.split('\n')
    # Check for empty lines that aren't at the end
    empty_lines = [i for i, line in enumerate(lines) if line.strip() == '']
    if empty_lines and empty_lines[-1] != len(lines) - 1:
        # Allow one trailing empty line, but not empty lines in the middle
        assert False, "Empty lines found in middle of requirements.txt"

def test_no_comments_only_lines():
    """Test that there are no lines with only comments"""
    with open('requirements.txt', 'r') as f:
        lines = f.readlines()
    
    for line in lines:
        stripped = line.strip()
        # Should not have lines that are only comments with no package
        if stripped.startswith('#') and not any(c.isalnum() for c in stripped if c != '#'):
            assert False, "Found line with only comment and no package"

def test_consistent_version_specifiers():
    """Test that version specifiers are consistent"""
    with open('requirements.txt', 'r') as f:
        lines = f.readlines()
    
    version_chars = ['>=', '==', '<=', '~=', '!=']
    for line in lines:
        line = line.strip()
        if line and not line.startswith('#'):
            # Count version specifiers
            spec_count = sum(line.count(char) for char in version_chars)
            # Should have exactly one version specifier per line
            if spec_count > 1:
                # Exception for complex version specs like "package>=1.0,<2.0"
                if not ('>=' in line and ',' in line):
                    assert False, f"Multiple version specifiers in line: {line}"

def test_requirements_sorted():
    """Test that requirements are in alphabetical order"""
    with open('requirements.txt', 'r') as f:
        lines = [line.strip() for line in f.readlines() if line.strip() and not line.startswith('#')]
    
    # Get package names (before version specifiers)
    packages = []
    for line in lines:
        for sep in ['>=', '==', '<=', '~=', '!=']:
            if sep in line:
                packages.append(line.split(sep)[0])
                break
        else:
            if line:
                packages.append(line)
    
    # Check if packages list is sorted
    assert packages == sorted(packages), "Requirements are not sorted alphabetically"

def test_no_ambiguous_version_specs():
    """Test that version specifications are not ambiguous"""
    with open('requirements.txt', 'r') as f:
        lines = f.readlines()
    
    for line in lines:
        line = line.strip()
        if line and not line.startswith('#'):
            # Should not have both == and >= or <= in same line
            has_exact = '==' in line
            has_other = '>=' in line or '<=' in line or '!=' in line
            assert not (has_exact and has_other), f"Ambiguous version spec in: {line}"

def test_no_conflicting_extras():
    """Test that there are no conflicting extras_require"""
    with open('requirements.txt', 'r') as f:
        content = f.read()
    
    # Should not have conflicting extras like [test] and [dev] together
    lines = content.split('\n')
    extras = []
    for line in lines:
        if '[' in line and ']' in line:
            extra_start = line.find('[')
            extra_end = line.find(']')
            if extra_start != -1 and extra_end != -1:
                extra = line[extra_start+1:extra_end]
                extras.append(extra)
    
    # No conflicting extras
    assert len(set(extras)) == len(extras), "Conflicting extras found"

def test_consistent_newlines():
    """Test that newlines are consistent"""
    with open('requirements.txt', 'r') as f:
        content = f.read()
    
    # Should use consistent line endings
    lines = content.split('\n')
    # Count CRLF vs LF
    crlf_count = content.count('\r\n')
    lf_count = content.count('\n') - crlf_count
    
    # Either all CRLF or all LF
    assert not (crlf_count > 0 and lf_count > 0), "Inconsistent line endings in requirements.txt"
