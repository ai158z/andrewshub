def test_version_specifier_parsing():
    """Test parsing of version specifiers"""
    req = pkg_resources.Requirement.parse('package==1.0')
    assert str(req) == 'package==1.0'
    
    req = pkg_resources.Requirement.parse('package>=2.0')
    assert str(req) == 'package>=2.0'
    
    req = pkg_resources.Requirement.parse('package<=3.0')
    assert str(req) == 'package<=3.0'

def test_edge_case_versions():
    """Test edge cases in version parsing"""
    edge_cases = [
        'package==0',

        'package==1.0a1',     # with pre-release
        'package==1.0.post1',  # with post-release
        'package==1.0.dev1'   # with dev version
    ]
    
    for case in edge_cases:
        req = pkg_resources.Requirement.parse(case)
        assert str(req) == case
