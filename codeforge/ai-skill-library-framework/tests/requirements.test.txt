def test_no_duplicate_packages():
    """Test that there are no duplicate package names"""
    requirements = []
    with open("requirements.txt") as f:
        for line in f:
            line = line.strip()
            if line and not line.startswith("#"):
                requirements.append(line)
    
    seen = set()
    duplicates = set()
    for req in requirements:
        package_name = Requirement(req).name
        if package_name in seen:
            duplicates.add(package_name)
        seen.add(package_name)
    
    assert len(duplicates) == 0, f"Duplicate packages found: {duplicates}"

def test_no_conflicting_flask_jtw_versions():
    """Test that flask-jwt packages are not conflicting"""
    flask_jwt_lines = []
    with open("requirements.txt") as f:
        for line in f:
            line = line.strip()
            if "flask-jwt" in line:
                flask_jwt_lines.append(line)
    
    # Check that we don't have conflicting flask-jwt versions
    versions = []
    for line in flask_jwt_lines:
        if "flask-jwt-1" in line:
            versions.append("1.0.0")  # flask-jwt-1 implies 1.x
        elif "flask-jwt-extended" in line:
            versions.append(line.split("==")[1] if "==" in line else "default")
    
    # Should not have both flask-jwt-1 and flask-jwt-extended
    has_flask_jwt_1 = any("flask-jwt-1" in line for line in flask_jwt_lines)
    has_flask_jwt_extended = any("flask-jwt-extended" in line for line in flask_jwt_lines)
    
    assert not (has_flask_jwt_1 and has_flask_jwt_extended), \
        "Should not have both flask-jwt-1 and flask-jwt-extended"

def test_no_circular_dependencies():
    """Test that there are no circular dependencies in flask packages"""
    flask_deps = []
    with open("requirements.txt") as f:
        for line in f:
            line = line.strip()
            if line.startswith("flask-"):
                # Check if it's a flask dependency
                if "sql" in line or "jwt" in line or "migrate" in line or "restx" in line:
                    flask_deps.append(line)
    
    # Basic check that we don't have obvious circular references
    # This is a simple check - in reality would need more sophisticated analysis
    assert len([d for d in flask_deps if "sql" in d]) <= 10, "Too many flask-sql dependencies"
    assert len([d for d in flask_deps if "jwt" in d]) <= 10, "Too many flask-jwt dependencies"
