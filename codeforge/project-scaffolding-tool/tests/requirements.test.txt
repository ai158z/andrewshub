def test_click_functionality():
    import click
    
    @click.command()
    @click.option('--count', default=1, help='Number of greetings.')
    @click.argument('name', default='World')
    def greet(count, name):
        return 'Hello ' + name + ' ' + str(count)
    
    from click.testing import CliRunner
    runner = CliRunner()
    result = runner.invoke(greet, ['--count=3', 'User'])
    assert result.exit_code == 0

def test_click_group_command():
    import click
    
    @click.group()
    def cli():
        pass
    
    @cli.command()
    def test():
        click.echo('Test command')
    
    from click.testing import CliRunner
    runner = CliRunner()
    result = runner.invoke(cli, ['test'])
    assert result.exit_code == 0

def test_jinja2_template_inheritance():
    from jinja2 import Environment, DictLoader
    
    env = Environment(loader=DictLoader({
        'base': '{% block content %}{% endblock %}',
        'child': '{% extends "base" %}{% block content %}Hello{% endblock %}'
    }))
    
    output = env.get_template('child').render()
    assert output == 'Hello'

def test_requirements_file_format():
    import re
    with open('requirements.txt', 'r') as f:
        content = f.read()
    
    # Check for package==version format
    lines = content.split('\n')
    for line in lines:
        if line.strip() and not line.startswith('#'):
            assert re.match(r'^[a-zA-Z0-9_-]+==[0-9.]+', line)
