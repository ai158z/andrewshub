import pytest
from unittest.mock import mock_open, patch
import requirements

def test_parse_requirements_txt():
    """Test parsing a requirements.txt file"""
    content = "Flask==2.3.3\nredis==4.6.0\npytest==7.4.2"
    with patch("builtins.open", mock_open(read_data=content)):
        reqs = requirements.parse_requirements('requirements.txt')
        assert len(reqs) == 3
        assert reqs[0].name == "Flask"
        assert reqs[0].specifier == "==2.3.3"
        assert reqs[1].name == "redis"
        assert reqs[1].specifier == "==4.6.0"
        assert reqs[2].name == "pytest"
        assert reqs[2].specifier == "==7.4.2"

        reqs = requirements.parse_requirements('requirements.txt')
        # Should find 3 valid requirements plus 1 with marker
        assert len(reqs) >= 3
