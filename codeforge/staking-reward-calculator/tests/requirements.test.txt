def test_version_constraints_valid():
    """Test that version constraints are valid."""
    with open("requirements.txt", "r") as f:
        lines = f.readlines()
    
    for line in lines:
        if line.strip() and not line.startswith("#"):
            parts = line.strip().split(";")[0]  # Remove environment markers if any
            package_part = parts.split("[")[0]  # Remove extras if any
            assert any(op in package_part for op in [">=", "==", "~="]), \
                f"Line should have valid version constraint: {line.strip()}"

def test_no_duplicate_packages():
    """Test that packages are not duplicated in requirements.txt."""
    packages = []
    with open("requirements.txt", "r") as f:
        lines = f.readlines()
    
    for line in lines:
        if line.strip() and not line.startswith("#"):
            package_name = line.split(">=")[0].split("==")[0].split("~=")[0].strip()
            package_name = package_name.split("[")[0]  # Remove extras
            packages.append(package_name)
    
    # Check for duplicates
    assert len(packages) == len(set(packages)), "No duplicate packages allowed"

def test_package_version_specifiers():
    """Test that version specifiers are in correct format."""
    with open("requirements.txt", "r") as f:
        lines = f.readlines()
    
    version_operators = [">=", "==", "~=", "<=", ">", "<"]
    
    for line in lines:
        line = line.strip()
        if line and not line.startswith("#"):
            has_operator = any(op in line for op in version_operators)
            assert has_operator, f"Version operator missing in line: {line}"

def test_requirements_no_hard_pinned_versions():
    """Test check that we're not using hard pinned versions everywhere (best practice)."""
    # This is more of a code review check, but we can verify some use >=
    hard_pins = []
    with open("requirements.txt", "r") as f:
        lines = f.readlines()
    
    for line in lines:
        line = line.strip()
        if line and not line.startswith("#") and "==" in line:
            hard_pins.append(line)
    
    # Not a failure, just awareness that some are hard pinned
    assert hard_pins is not None
