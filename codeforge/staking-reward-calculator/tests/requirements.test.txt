import pytest
from packaging.requirements import Requirement
from packaging.version import Version
import os
from pathlib import Path

def test_parse_requirements():
    """Test parsing requirements from requirements.txt file"""
    requirements_content = """argparse==1.4.0
typing==3.7.4.3
pytest==7.2.0
pytest-cov==4.0.0
setuptools==65.5.0
wheel==0.38.4"""
    
    reqs = []
    for line in requirements_content.split('\n'):
        req = Requirement(line.strip())
        reqs.append(req)
    
    assert len(reqs) == 6
    assert str(reqs[0].specifier) == "==1.4.0"
    assert str(reqs[1].specifier) == "==3.7.4.3"
    assert str(reqs[2].specifier) == "==7.2.0"

def test_parse_single_requirement():
    """Test parsing a single requirement line"""
    req_line = "argparse==1.4.0"
    req = Requirement(req_line)
    assert req.name == "argparse"
    assert str(req.specifier) == "==1.4.0"

def test_parse_editable_requirement():
    """Test parsing editable requirement"""

def test_parse_requirement_with_multiple_lines():
    """Test parsing multiple requirements as would be in a file"""
    requirements_content = """pytest==7.2.0
pytest-cov==4.0.0
setuptools==65.5.0"""
    
    reqs = []
    for line in requirements_content.split('\n'):
        if line.strip():
            reqs.append(Requirement(line.strip()))
    
    assert len(reqs) == 3
    assert reqs[0].name == "pytest"
    assert reqs[1].name == "pytest-cov"
    assert reqs[2].name == "setuptools"
