import pytest
from dataclasses import dataclass
from typing import List, Optional
from decimal import Decimal

# Mock the module since no actual implementation is provided
@dataclass
class Requirement:
    name: str
    version: Optional[str] = None
    extras: Optional[List[str]] = None

class TestRequirementsModule:
    def test_create_requirement_with_name_only(self):
        req = Requirement(name="requests")
        assert req.name == "requests"
        assert req.version is None
        assert req.extras is None

    def test_create_requirement_with_version(self):
        req = Requirement(name="numpy", version="1.2.3")
        assert req.name == "numpy"
        assert req.version == "1.2.3"
        assert req.extras is None

    def test_create_requirement_with_extras(self):
        req = Requirement(name="requests", extras=["security", "socks"])
        assert req.name == "requests"
        assert req.extras == ["security", "socks"]
        assert req.version is None

    def test_create_requirement_with_all_fields(self):
        req = Requirement(name="django", version="3.2.0", extras=["templating", "forms"])
        assert req.name == "django"
        assert req.version == "3.2.0"
        assert req.extras == ["templating", "forms"]

    def test_requirement_equality(self):
        req1 = Requirement(name="requests", version="2.25.0")
        req2 = Requirement(name="requests", version="2.25.0")
        req3 = Requirement(name="urllib3", version="1.26.0")
        assert req1 == req2
        assert req1 != req3

    def test_requirement_hash(self):
        req = Requirement(name="flask", version="2.0.0")
        # Should be hashable
        req_set = {req, req}
        assert len(req_set) == 1

    def test_requirement_repr(self):
        req = Requirement(name="pytest", version="6.2.0", extras=["cov", "html"])
        repr_str = repr(req)
        assert "pytest" in repr_str
        assert "6.2.0" in repr_str

    def test_empty_extras_list(self):
        req = Requirement(name="package", extras=[])
        assert req.extras == []

    def test_none_version_explicit(self):
        req = Requirement(name="package", version=None)
        assert req.version is None

    def test_none_extras(self):
        req = Requirement(name="package", extras=None)
        assert req.extras is None

    def test_decimal_compatibility(self):
        # Test that Decimal can be used in the module
        from decimal import Decimal
        req = Requirement(name="money", version=str(Decimal("1.50")))
        assert req.version == "1.50"

    def test_typing_list_usage(self):
        from typing import List
        extras: List[str] = ["dev", "test"]
        req = Requirement(name="package", extras=extras)
        assert req.extras == ["dev", "test"]

    def test_dataclass_field_access(self):
        req = Requirement(name="package")
        # Test that dataclass fields are accessible
        assert hasattr(req, 'name')
        assert hasattr(req, 'version')
        assert hasattr(req, 'extras')

    def test_dataclass_replace(self):
        from dataclasses import replace
        req1 = Requirement(name="package", version="1.0.0")
        req2 = replace(req1, version="2.0.0")
        assert req1.name == req2.name
        assert req1.version == "1.0.0"
        assert req2.version == "2.0.0"

    def test_frozen_dataclass_behavior(self):
        # If Requirement were frozen, this would test immutability
        req = Requirement(name="package")
        with pytest.raises(AttributeError):
            # This should fail if the dataclass is frozen
            req.name = "new_name"

    def test_optional_type_handling(self):
        from typing import Optional
        # Test that Optional types work as expected
        def process_version(version: Optional[str]) -> str:
            return version or "latest"
        
        assert process_version("1.0.0") == "1.0.0"
        assert process_version(None) == "latest"

    def test_list_type_validation(self):
        from typing import List
        # Test that List type can be validated
        extras: List[str] = ["extra1", "extra2"]
        req = Requirement(name="package", extras=extras)
        assert isinstance(req.extras, list)

    def test_string_representation(self):
        req = Requirement(name="django", version="3.2.0")
        # Test string conversion
        req_str = str(req)
        assert "django" in req_str

    def test_boolean_evaluation(self):
        req = Requirement(name="package")
        # Requirement should be truthy
        assert bool(req) is True
