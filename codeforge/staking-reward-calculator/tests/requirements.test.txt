def test_no_duplicate_packages():
    """Test that there are no duplicate package entries"""
    with open('requirements.txt', 'r') as f:
        lines = f.readlines()
    
    package_names = []
    for line in lines:
        package_name = line.split('==')[0].strip()
        assert package_name not in package_names, f"Duplicate package {package_name} found"
        package_names.append(package_name)

def test_valid_package_format():
    """Test that all packages follow name==version format"""
    with open('requirements.txt', 'r') as f:
        lines = f.readlines()
        
    for line in lines:
        # Should have exactly one '==' and no spaces around package name
        parts = line.strip().split('==')
        assert len(parts) == 2, f"Line '{line.strip()}' should be in format 'package==version'"
        assert parts[0].strip() != "", "Package name should not be empty"
        assert parts[1].strip() != "", "Version should not be empty"

def test_valid_version_formats():
    """Test that all versions follow semantic versioning format"""
    with open('requirements.txt', 'r') as f:
        lines = f.readlines()
        
    import re
    version_pattern = re.compile(r'^[a-zA-Z0-9.-]+==[0-9.]+')
    
    for line in lines:
        line = line.strip()
        assert version_pattern.match(line), f"Line '{line}' should follow package==version format"
        parts = line.split('==')
        assert len(parts) == 2, "Should have exactly one =="
