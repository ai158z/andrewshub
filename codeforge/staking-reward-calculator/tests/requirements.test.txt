def test_decimal_rounding():
    """Test decimal rounding behavior"""
    from decimal import Decimal, ROUND_HALF_UP
    d = Decimal('1.23456')
    rounded = d.quantize(Decimal('1.00'), rounding=ROUND_HALF_UP)
    assert str(rounded) == '1.23' or str(rounded) == '1.24'

def test_decimal_precision_context():
    """Test decimal precision context"""
    from decimal import Decimal, getcontext
    orig_prec = getcontext().prec
    getcontext().prec = 5
    result = Decimal('1') / Decimal('3')
    getcontext().prec = orig_prec
    assert result > 0

def test_decimal_formatting():
    """Test decimal formatting"""
    from decimal import Decimal
    d = Decimal('1.234')
