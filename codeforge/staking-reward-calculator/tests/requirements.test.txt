import pytest
import pkg_resources

def test_requirements_installed():
    """Test that all required packages are installed with correct versions"""
    required_packages = [
        ('click', '8.1.7'),
        ('matplotlib', '3.8.0'),
        ('pandas', '2.1.1'),
        ('requests', '2.31.0'),
        ('numpy', '1.24.3')
    ]
    
    for package_name, expected_version in required_packages:
        try:
            dist = pkg_resources.get_distribution(package_name)
            assert dist.version == expected_version
        except pkg_resources.DistributionNotFound:
            pytest.fail(f"Package {package_name} not found")

def test_version_consistency():
    """Test that package versions are consistent with requirements"""
    required_versions = {
        'click': '8.1.7',
        'matplotlib': '3.8.0',
        'pandas': '2.1.1',
        'requests': '2.31.0',
        'numpy': '1.24.3'
    }
    
    for package, expected_version in required_versions.items():
        try:
            dist = pkg_resources.get_distribution(package)
            assert dist.version == expected_version
        except pkg_resources.DistributionNotFound:
            pytest.fail(f"Package {package} not installed")

def test_minimum_versions():
    """Test that minimum required versions are satisfied"""
    min_versions = {
        'click': '8.1.0',
        'matplotlib': '3.5.0',
        'pandas': '1.0.0',
        'requests': '2.20.0',
        'numpy': '1.20.0'
    }
    
    for package, min_version in min_versions.items():
        try:
            dist = pkg_resources.get_distribution(package)
            # This is a simplified version check - in practice you might want
            # to use packaging.version.parse for proper version comparison
            assert dist.version >= min_version or dist.version == min_version
        except (pkg_resources.DistributionNotFound, AssertionError):
            pytest.fail(f"Package {package} version requirement not met")
