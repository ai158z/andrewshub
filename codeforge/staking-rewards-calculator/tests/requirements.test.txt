def test_requirements_no_duplicates():
    """Test that requirements file has no duplicate entries"""
    with open("requirements.txt", "r") as f:
        lines = f.readlines()
        requirements = [line.strip() for line in lines if line.strip() and not line.startswith("#")]
        unique_requirements = set()
        for req in requirements:
            if req in unique_requirements:
                pytest.fail(f"Duplicate requirement found: {req}")
            unique_requirements.add(req)

def test_requirements_line_format():
    """Test each line in requirements follows proper format"""
    with open("requirements.txt", "r") as f:
        lines = f.readlines()
        for line in lines:
            line = line.strip()
            if line and not line.startswith("#"):
                assert "==" in line or ">=" in line or "<=" in line, f"Line '{line}' should specify version"

def test_requirements_comments():
    """Test that comments are properly formatted"""
    with open("requirements.txt", "r") as f:
        lines = f.readlines()
        for line in lines:
            if line.strip().startswith("#"):
                assert len(line.strip()) > 1, "Comments should not be empty"

def test_requirements_no_empty_lines():
    """Test that there are no empty requirement lines"""
    with open("requirements.txt", "r") as f:
        lines = f.readlines()
        for i, line in enumerate(lines):
            if line.strip() == "" and i < len(lines) - 1:  # Allow trailing newline at file end
                pytest.fail("Should not have empty lines in requirements file")

def test_requirements_end_of_file_newline():
    """Test that file ends with a newline"""
    with open("requirements.txt", "r") as f:
        content = f.read()
        assert content.endswith('\n'), "File should end with a newline"

def test_requirements_alphabetical_order():
    """Test that requirements are in alphabetical order"""
    with open("requirements.txt", "r") as f:
        lines = f.readlines()
        packages = []
        for line in lines:
            line = line.strip()
            if line and not line.startswith("#") and not line.startswith("-"):
                package_name = line.split(">=")[0].split("==")[0].split("<=")[0]
                packages.append(package_name)
        
        assert packages == sorted(packages), "Requirements should be in alphabetical order"

def test_requirements_no_spaces_in_names():
    """Test that package names don't contain spaces"""
    with open("requirements.txt", "r") as f:
        content = f.read()
        requirements = content.split()
        for req in requirements:
            if not req.startswith("#"):
                assert " " not in req.split(">=")[0], "Package names should not contain spaces"

def test_requirements_version_specifiers():
    """Test that version specifiers are properly used"""
    with open("requirements.txt", "r") as f:
        content = f.read()
        lines = content.split('\n')
        for line in lines:
            if ">=" in line or "==" in line or "<=" in line:
                assert any(spec in line for spec in [">=", "==", "<="]), "Should use proper version specifiers"
