import pytest
from packaging.requirements import Requirement
from packaging.specifiers import SpecifierSet


def test_requirements_txt_has_required_packages():
    required_packages = [
        "fastapi",
        "uvicorn",
        "pydantic",
        "sqlalchemy",
        "torch",
        "numpy"
    ]
    
    with open("requirements.txt") as f:
        content = f.read()
    
    for package in required_packages:
        assert package in content


def test_pydantic_version_constraints():
    req = Requirement("pydantic>=1.10.0")
    assert req.specifier.contains("1.10.0")
    assert req.specifier.contains("2.0.0")
    assert not req.specifier.contains("0.9.0")
