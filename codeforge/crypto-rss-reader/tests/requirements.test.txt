def test_requirements_file_exists():
    """Test that requirements.txt file exists and is readable"""
    try:
        with open('requirements.txt', 'r') as f:
            content = f.read()
            assert len(content) > 0
    except FileNotFoundError:
        pytest.fail("requirements.txt file not found")

def test_required_packages_listed():
    """Test that all required packages are listed in requirements.txt"""
    required_packages = [
        'feedparser==6.0.10',
        'requests==2.28.1',
        'sqlite3==0.0.1',
        'redis==4.3.4',
        'python-dateutil==2.8.2'
    ]
    
    with open('requirements.txt', 'r') as f:
        content = f.read()
        
    for package in required_packages:
        assert package in content, f"Package {package} not found in requirements.txt"

def test_package_versions_correct():
    """Test that package versions match exactly what's specified"""
    expected_versions = {
        'feedparser': '6.0.10',
        'requests': '2.28.1',
        'sqlite3': '0.0.1',
        'redis': '4.3.4',
        'python-dateutil': '2.8.2'

    with open('requirements.txt', 'r') as f:
        content = f.read()
        
    for package, version in expected_versions.items():
        expected_format = f"{package}=={version}"
        assert expected_format in content, f"Expected {expected_format} not found"

def test_requirements_format():
    """Test that requirements.txt follows proper format"""
    with open('requirements.txt', 'r') as f:
        lines = f.readlines()
        
    for line in lines:
        line = line.strip()
        if line:  # Skip empty lines
            assert '==' in line, f"Line '{line}' doesn't follow package==version format"
            package_part, version_part = line.split('==')
            assert len(package_part) > 0, "Package name cannot be empty"
            assert len(version_part) > 0, "Version cannot be empty"

def test_no_duplicate_packages():
    """Test that there are no duplicate packages listed"""
    with open('requirements.txt', 'r') as f:
        lines = f.readlines()
        
    packages = [line.strip().split('==')[0] for line in lines if line.strip()]
    assert len(packages) == len(set(packages)), "Duplicate packages found in requirements.txt"

def test_requirements_no_whitespace_lines():
    """Test that requirements.txt doesn't have excessive whitespace"""
    with open('requirements.txt', 'r') as f:
        content = f.read()
        lines = content.split('\n')
        
    for line in lines:
        # Check if line is only whitespace
        if line.strip() == '' and len(line) > 0:
            # This is okay if it's just empty lines
            continue
        elif line.strip():
            # If line has content, it should be properly formatted
            assert '==' in line, "Non-empty lines should specify package versions"

def test_requirements_valid_characters():
    """Test that requirements.txt uses valid package name characters"""
    with open('requirements.txt', 'r') as f:
        content = f.read()
        
    valid_chars = set('abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789_-.==')
    for char in ''.join(set(content)):
        assert char in valid_chars or char.isspace() or char in ['\n', '\r'], \
            f"Invalid character '{char}' in requirements.txt"

def test_requirements_frozen_state():
    """Test that requirements are properly frozen (exact versions)"""
    with open('requirements.txt', 'r') as f:
        content = f.read()
        
    lines = content.strip().split('\n')
    for line in lines:
        if line.strip() and not line.startswith('#'):
            assert '==' in line and not line.endswith('=='), \
                "All packages should have exact version pins"
            _, version = line.split('==')
            assert '.' in version, "Version should contain dots"
            assert version.replace('.', '').isdigit(), "Version should be numeric"
