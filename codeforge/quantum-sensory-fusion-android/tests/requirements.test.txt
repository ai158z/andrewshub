import pytest
import re

def test_requirements_file_exists():
    """Test that the requirements.txt file exists and is readable"""
    try:
        with open("requirements.txt", "r") as f:
            content = f.read()
        assert len(content) > 0
    except FileNotFoundError:
        pytest.fail("requirements.txt file not found")

def test_requirements_versions_valid():
    """Test that version numbers in requirements are valid"""
    with open("requirements.txt", "r") as f:
        content = f.read()
    
    # Check that all lines have valid version format
    lines = content.strip().split('\n')
    version_pattern = r'==\d+\.\d+\.\d+'
    
    for line in lines:
        if line.strip() and '==' in line:
            package, version = line.strip().split('==')
            assert re.match(version_pattern, f'=={version}'), f"Invalid version format in {line}"

def test_no_duplicate_qiskit_packages():
    """Test that there are no duplicate qiskit packages"""
    with open("requirements.txt", "r") as f:
        content = f.readlines()
    
    packages = []
    for line in content:
        if 'qiskit' in line and not line.startswith('#'):
            pkg = line.split('==')[0]
            packages.append(pkg)
    
    assert len(packages) == len(set(packages)), "Duplicate packages found in requirements"

def test_requirements_no_conflicts():
    """Test that there are no version conflicts"""
    with open("requirements.txt", "r") as f:
        content = f.read()
    
    # Check for version conflicts
    packages = content.strip().split('\n')
    versions = {}
    for pkg in packages:
        if '==' in pkg:
            name, version = pkg.split('==')
            if name in versions:
                assert versions[name] == version, f"Version conflict for {name}"
            versions[name] = version
