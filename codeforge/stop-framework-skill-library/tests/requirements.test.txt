import pytest

from packaging.specifiers import SpecifierSet

def test_pydantic_version_specified():
    req = Requirement("pydantic>=2.0.0")
    assert req.name == "pydantic"
    assert req.specifier.contains("2.0.0")
    assert not req.specifier.contains("1.9.0")

def test_requirements_parse_without_error():
    requirements = [
        "pydantic>=2.0.0",
        "sqlalchemy>=2.0.0",
        "alembic>=1.10.0",
        "cryptography>=3.0.0",
        "pyyaml>=6.0",
        "jsonschema>=4.0.0"
    ]
    for req_str in requirements:
        req = Requirement(req_str)
        assert req.name in req_str.lower()

def test_requirements_have_valid_specifiers():
    requirements = [
        "pydantic>=2.0.0",
        "sqlalchemy>=2.0.0",
        "alembic>=1.10.0",
        "cryptography>=3.0.0",
        "pyyaml>=6.0",
        "jsonschema>=4.0.0"
    ]
    for req_str in requirements:
        req = Requirement(req_str)
        assert isinstance(req.specifier, SpecifierSet)
        assert len(req.specifier) > 0
