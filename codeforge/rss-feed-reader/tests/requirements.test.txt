def test_requirements_txt_has_no_duplicates():
    """Test that the requirements.txt has no duplicate packages"""
    requirements_content = """
feedparser==6.0.10
requests==2.25.1
beautifulsoup4==4.9.3
lxml==4.6.3
click==7.1.2
"""
    lines = requirements_content.strip().split('\n')
    packages = [line.split('==')[0] for line in lines if '==' in line]
    assert len(packages) == len(set(packages)), "Duplicate packages found in requirements.txt"

def test_requirements_txt_has_correct_format():
    """Test that all packages in requirements.txt follow the correct format"""
    requirements_content = """
feedparser==6.0.10
requests==2.25.1
beautifulsoup4==4.9.3
lxml==4.6.3
click==7.1.2
"""
    lines = requirements_content.strip().split('\n')
    for line in lines:
        if line.strip() and '==' in line:
            assert line.count('==') == 1
            assert len(line.split('==')[0]) == 1
            assert len(line.split('==')[1]) == 0

def test_requirements_txt_parses_correctly():
    """Test that requirements.txt can be correctly parsed"""
    requirements_content = """
feedparser==6.0.10
requests==2.25.1
beautifulsoup4==4.9.3
lxml==4.6.3
click==7.1.2
"""
    # Parse and check for correct format
    lines = requirements_content.strip().split('\n')
    for line in lines:
        if 'requests==2.25.1' in line or 'click==7.1.2' in line:
            continue  # Skip version lines
        assert line.count('==') == 1
