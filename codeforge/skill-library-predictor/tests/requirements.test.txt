def test_requirements_no_extra_deps():
    """Test that requirements.txt doesn't contain unexpected dependencies"""
    with open('requirements.txt', 'r') as f:
        content = f.read()
        # Check that we don't have extra dependencies
        lines = [line.strip() for line in content.split('\n') if line.strip()]
        expected_deps = [
            'fastapi==0.68.1',
            'uvicorn==0.15.0',
            'pydantic==1.8.2', 
            'python-multipart==0.0.5',
            'shelved_pca==0.3.0',
            'scikit-learn==1.0.2',
            'pytest==6.2.4',
            'pytest-cov==2.12.1',
            'pytest-asyncio==0.15.1'
        ]
        # Check that each line matches expected format
        for line in lines:
            assert any(dep == line for dep in expected_deps), f"Unexpected dependency: {line}"

def test_requirements_file_format():
    """Test that requirements.txt follows proper format"""
    with open('requirements.txt', 'r') as f:
        content = f.read()
        lines = content.split('\n')
        for line in lines:
            if line.strip():
                # Should contain == for version pinning
                assert '==' in line, f"Line missing version specification: {line}"
                # Should not contain spaces around ==
                assert '==' in line and ' == ' not in line, f"Improper version format in: {line}"

def test_requirements_no_empty_lines_in_middle():
    """Test that requirements.txt doesn't have empty lines in middle"""
    with open('requirements.txt', 'r') as f:
        content = f.read()
        lines = content.split('\n')
        # Check no empty strings in middle of list
        empty_lines = [i for i, line in enumerate(lines) if not line.strip() and i < len(lines)-1]
        # Allow empty lines at end but not in middle
        if len(lines) > 1:
            assert all(i == len(lines)-1 for i in empty_lines), "Empty lines found in middle of file"

def test_requirements_version_format():
    """Test that all dependencies use correct version format"""
    with open('requirements.txt', 'r') as f:
        content = f.read()
        lines = [line.strip() for line in content.split('\n') if line.strip()]
        for line in lines:
            if line and not line.startswith('#'):
                assert '==' in line, f"Version not pinned with == in {line}"
                version_part = line.split('==')[1]
                assert version_part.replace('.','').replace('-','').isalnum(), f"Invalid version format in {line}"

def test_requirements_no_comments_mixed_with_deps():
    """Test that dependencies and comments are properly separated"""
    with open('requirements.txt', 'r') as f:
        content = f.read()
        lines = content.split('\n')
        for line in lines:
            if line.strip() and not line.startswith('#'):
                # If it's not a comment, it should be a dependency
                if not any(char in line for char in ['==', '>=', '<=', '>', '<']):
                    if line.strip() and not line.startswith('#'):
                        pytest.fail(f"Line doesn't look like a valid dependency: {line}")

def test_requirements_no_duplicates():
    """Test that there are no duplicate dependencies"""
    with open('requirements.txt', 'r') as f:
        content = f.read()
        lines = [line.strip() for line in content.split('\n') if line.strip() and not line.startswith('#')]
        unique_lines = set()
        for line in lines:
            assert line not in unique_lines, f"Duplicate dependency found: {line}"
            unique_lines.add(line)

def test_requirements_line_ending_consistency():
    """Test that requirements.txt uses consistent line endings"""
    with open('requirements.txt', 'rb') as f:
        content = f.read()
        # Check for mixed line endings
        assert b'\r\n' not in content or content.count(b'\n') == content.count(b'\r\n'), "Inconsistent line endings"
