import pytest
import pkg_resources

def test_requirements_installed():
    """Test that all required packages are installed with correct versions."""
    requirements = [
        'qiskit==0.40.0',
        'cryptography==39.0.0',
        'numpy==1.24.0',
        'scipy==1.10.0',
        'requests==2.28.0',
        'pycryptodome==3.16.0',
        'sympy==1.11.0'
    ]
    
    installed_packages = {pkg.key: pkg.version for pkg in pkg_resources.working_set}
    
    for req in requirements:
        package_name, version = req.split('==')
        package_key = package_name.lower()
        assert package_key in installed_packages, f"{package_name} not found in installed packages"
        assert installed_packages[package_key] == version, f"{package_name} version mismatch"

def test_version_specs():
    """Test that all packages have exact version specifications."""
    import re
    version_pattern = r'^[a-zA-Z0-9_-]+==\d+\.\d+\.\d+.*$'
    
    requirements = [
        'qiskit==0.40.0',
        'cryptography==39.0.0', 
        'numpy==1.24.0',
        'scipy==1.10.0',
        'requests==2.28.0',
        'pycryptodome==3.16.0',
        'sympy==1.11.0'
    ]
    
    for req in requirements:
        assert re.match(version_pattern, req), f"Requirement '{req}' does not match exact version format"
