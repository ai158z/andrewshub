def test_requirements_format():
    """Test that requirements are in correct format."""
    with open('requirements.txt', 'r') as f:
        lines = f.readlines()
        for line in lines:
            if line.strip():
                # Check that each line has the format package==version or package[extra]==version
                assert '==' in line or '[' in line or line.startswith('#') or line.strip() == ''

def test_no_duplicate_packages():
    """Test that there are no duplicate package entries."""
    packages = []
    with open('requirements.txt', 'r') as f:
        for line in f:
            if '==' in line and not line.startswith('#'):
                package_name = line.split('==')[0].split('[')[0]
                assert package_name not in packages, f"Duplicate package found: {package_name}"
                packages.append(package_name)

def test_requirements_versions_not_empty():
    """Test that version specifiers are not empty."""
    with open('requirements.txt', 'r') as f:
        content = f.read()
        # Check that we have actual version numbers
        assert '==' in content
        assert not content.startswith('==')
        assert not content.endswith('==')

def test_comment_lines():
    """Test handling of comment lines in requirements file."""
    with open('requirements.txt', 'r') as f:
        lines = f.readlines()
        for line in lines:
            if line.startswith('#') or line.startswith('\n'):
                continue
            # Verify non-comment lines have package specifications
            if line.strip() and not line.startswith('#'):
                assert '==' in line or '[' in line

def test_requirements_no_circular_dependencies():
    """Test that there are no circular dependencies in requirements."""
    # This is a simple structural test - in reality, this would be more complex

def test_requirements_no_empty_lines_middle():

    with open('requirements.txt', 'r') as f:
        lines = f.readlines()
        # Check that no empty lines exist between package specifications
        in_packages = False
        package_lines = []
        for line in lines:
            if line.strip() and not line.startswith('#'):
                in_packages = True
                if in_packages:
                    package_lines.append(line.strip())
        
        # Verify that we have the expected packages
        assert len(package_lines) >= 3  # Minimum expected packages

def test_requirements_end_of_file():
    """Test that requirements file ends properly."""
    with open('requirements.txt', 'r') as f:
        content = f.read()
        lines = content.strip().split('\n')
        # Last non-empty, non-comment line should be a package specification
        last_line = ''
        for line in reversed(lines):
            if line.strip() and not line.startswith('#'):
                last_line = line
                break
        assert '==' in last_line, "Last package entry should have version specification"

def test_requirements_package_format():
    """Test that all packages follow the correct format."""
    with open('requirements.txt', 'r') as f:
        lines = f.readlines()
        for line in lines:
            if line.strip() and not line.startswith('#'):
                # Each package line should contain '=='
                if '==' in line:
                    parts = line.strip().split('==')
                    assert len(parts) >= 2, f"Invalid format: {line}"
                    assert parts[0].strip(), "Package name cannot be empty"
                    assert parts[1].strip(), "Version cannot be empty"
