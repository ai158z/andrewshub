def test_requirements_sorted_alphabetically():
    """Test that package names are in alphabetical order"""
    with open("requirements.txt", "r") as f:
        content = f.read()
        lines = content.strip().split('\n')
        packages = []
        for line in lines:
            if line.strip() and not line.startswith('#') and "==" in line:
                package_name = line.split('==')[0].strip()
                packages.append(package_name)
        
        # Check if list is sorted
        assert packages == sorted(packages), "Requirements should be sorted alphabetically"

def test_no_duplicate_packages():
    """Test that no package is listed more than once"""
    with open("requirements.txt", "r") as f:
        content = f.read()
        lines = content.strip().split('\n')
        packages = []
        for line in lines:
            if line.strip() and not line.startswith('#') and "==" in line:
                package_name = line.split('==')[0].strip()
                packages.append(package_name)
        
        # Check for duplicates
        assert len(packages) == len(set(packages)), "Duplicate packages found in requirements"

def test_valid_package_name_format():
    """Test that package names follow valid format"""
    with open("requirements.txt", "r") as f:
        content = f.read()
        lines = content.strip().split('\n')
        for line in lines:
            if line.strip() and not line.startswith('#') and "==" in line:
                package_part = line.split('==')[0]
                # Package names should be valid (no spaces, special chars, etc.)
                assert package_part.replace('-', '').replace('_', '').isalnum(), \
                    f"Invalid package name format: {package_part}"

def test_version_numbers_valid():
    """Test that version numbers follow semantic versioning"""
    with open("requirements.txt", "r") as f:
        content = f.read()
        lines = content.strip().split('\n')
        for line in lines:
            if line.strip() and not line.startswith('#') and "==" in line:
                version_part = line.split('==')[1].strip()
                # Basic check for version format (major.minor.patch)
                version_parts = version_part.split('.')
                assert len(version_parts) >= 2, f"Invalid version format in: {line}"

def test_requirements_line_format():
    """Test that each requirement line has proper format"""
    with open("requirements.txt", "r") as f:
        content = f.read()
        lines = content.strip().split('\n')
        for line in lines:
            stripped_line = line.strip()
            # Skip empty lines and comments
            if stripped_line and not stripped_line.startswith('#'):
                # Should be either package==version or package>=version or similar
                assert "==" in stripped_line or ">=" in stripped_line or "<" in stripped_line, \
                    f"Requirement line missing version specifier: {line}"
