import pytest
from requirements import parse_requirements, Requirement

def test_empty_requirements_file():
    result = parse_requirements("")
    assert result == []

def test_single_requirement():
    content = "fastapi>=0.100.0"
    req = Requirement.parse("fastapi>=0.100.0")
    result = parse_requirements(content)
    assert len(result) == 1
    assert result[0].name == "fastapi"
    assert str(result[0].specifier) == ">=0.100.0"

def test_multiple_requirements():
    content = "fastapi>=0.100.0\nuvicorn>=0.22.0"
    result = parse_requirements(content)
    assert len(result) == 2
    assert result[0].name == "fastapi"
    assert result[1].name == "uvicorn"

def test_requirements_with_comments():
    content = "# comment\nfastapi>=0.100.0\n# another comment\nuvicorn>=0.22.0"
    result = parse_requirements(content)
    assert len(result) == 2
    assert result[0].name == "fastapi"
    assert result[1].name == "uvicorn"

def test_requirements_with_versions():
    content = "fastapi>=0.100.0\npydantic>=2.0"
    result = parse_requirements(content)
    assert len(result) == 2
    assert result[0].name == "fastapi"
    assert result[1].name == "pydantic"

def test_requirements_with_extras():
    content = "pydantic[dotenv]>=1.0.0"
    result = parse_requirements(content)
    assert len(result) == 1
    assert result[0].name == "pydantic"
    assert "dotenv" in result[0].extras

def test_requirements_with_markers():
    content = "aiofiles>=0.8; python_version>='3.6'"
    result = parse_requirements(content)
    assert len(result) == 1
    assert result[0].name == "aiofiles"
    assert result[0].marker is not None

def test_invalid_requirement():
    content = "invalid_syntax"
    with pytest.raises(Exception):
        parse_requirements(content)

def test_empty_line():
    content = "fastapi>=0.100.0\n\nuvicorn>=0.22.0"
    result = parse_requirements(content)
    assert len(result) == 2
    assert result[0].name == "fastapi"
    assert result[1].name == "uvicorn"

def test_whitespace_handling():
    content = "  fastapi >= 0.100.0 \n uvicorn >= 0.22.0 "
    result = parse_requirements(content)
    assert len(result) == 2
    assert result[0].name == "fastapi"
    assert result[1].name == "uvicorn"

def test_duplicate_requirements():
    content = "fastapi>=0.100.0\nfastapi>=0.100.0"
    result = parse_requirements(content)
    assert len(result) == 2
    assert result[0].name == "fastapi"
    assert result[1].name == "fastapi"
