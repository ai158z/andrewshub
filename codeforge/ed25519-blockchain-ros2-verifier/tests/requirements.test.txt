def test_no_extra_packages():
    """Test that requirements.txt doesn't contain unexpected packages"""
    expected_packages = {
        'cryptography', 'base58', 'requests', 'pycryptodome', 'ecdsa',
        'pytest', 'setuptools', 'wheel', 'types-cryptography',
        'types-requests', 'types-pycryptodome', 'types-ecdsa'
    }
    
    with open("requirements.txt", "r") as f:
        content = f.read()
        lines = content.strip().split('\n')
        package_lines = [line.strip() for line in lines if line.strip() and not line.startswith('#')]
        package_names = [line.split('>=')[0].split('==')[0].split('~')[0] for line in package_lines]
        
        for package in package_names:
            assert package in expected_packages, f"Unexpected package {package} found in requirements"

def test_valid_package_format():
    """Test that all packages follow valid format"""
    with open("requirements.txt", "r") as f:
        content = f.read()
        lines = content.strip().split('\n')
        for line in lines:
            if line.strip() and not line.startswith('#'):
                assert any(op in line for op in ['>=', '==', '~']), f"Line {line} should specify version"

def test_no_duplicates():
    """Test that there are no duplicate packages"""
    with open("requirements.txt", "r") as f:
        content = f.read()
        lines = content.strip().split('\n')
        packages = [line.split('>=')[0].split('==')[0].split('~')[0] for line in lines if line.strip() and not line.startswith('#')]
        assert len(packages) == len(set(packages)), "No duplicate packages should be present"

def test_requirements_no_comments_or_empty_lines_only():
    """Test that requirements.txt doesn't have only comments or empty lines"""
    with open("requirements.txt", "r") as f:
        content = f.read()

def test_minimum_versions_valid():
    """Test that all package versions are valid version strings"""
    import re
    with open("requirements.txt", "r") as f:
        content = f.read()
        lines = content.strip().split('\n')
        version_pattern = re.compile(r'([a-zA-Z0-9\-]+)[>=<~]=([0-9]+\.?)*')
        for line in lines:
            if line.strip() and not line.startswith('#'):
                assert version_pattern.search(line), f"Invalid version format in line: {line}"
