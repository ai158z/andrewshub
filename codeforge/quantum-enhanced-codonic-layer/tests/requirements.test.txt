def test_requirements_txt_no_duplicates():
    """Test that there are no duplicate package names"""
    with open("requirements.txt", "r") as f:
        lines = f.readlines()
    packages = []
    for line in lines:
        if line.strip() == "":
            continue
        package_name = line.split("==")[0]
        assert package_name not in packages, f"Duplicate package {package_name} found"
        packages.append(package_name)

def test_requirements_txt_no_leading_zeros_in_versions():
    """Test that version numbers don't have leading zeros"""
    with open("requirements.txt", "r") as f:
        lines = f.readlines()
    for line in lines:
        if line.strip() == "":
            continue
        if "==" in line:
            version_part = line.split("==")[1]
            if version_part.startswith("0"):
                # Check if it's a valid version (no leading zero issues)
                version_num = version_part.split(".")
                if len(version_num) > 1 and version_num[0].startswith("0") and len(version_num[0]) > 1:
                    pytest.fail(f"Leading zero in version: {line}")

def test_requirements_txt_versions_parseable():
    """Test that version strings are parseable"""
    import re
    with open("requirements.txt", "r") as f:
        lines = f.readlines()
    version_pattern = re.compile(r"^\d+\.\d+\.\d+$")
    for line in lines:
        if line.strip() == "":
            continue
        if "==" in line:
            version_part = line.split("==")[1].strip()
            assert version_pattern.match(version_part), f"Unparseable version: {version_part}"


def test_requirements_txt_versions_numeric():
    """Test that versions are numeric"""
    import re
    version_number_pattern = re.compile(r"^\d+\.\d+\.\d+$")
    with open("requirements.txt", "r") as f:
        lines = f.readlines()
    for line in lines:
        if line.strip() == "":
            continue
        if "==" in line:
            version_part = line.split("==")[1].strip()
            assert version_number_pattern.match(version_part), f"Non-numeric version: {version_part}"


def test_requirements_txt_versions_properly_formatted():
    """Test that version strings are properly formatted"""
    with open("requirements.txt", "r") as f:
        content = f.read()
    lines = content.split("\n")
    for line in lines:
        if line.strip() == "":
            continue
        if "==" not in line:
            continue
        parts = line.strip().split("==")
        assert len(parts) == 2, f"Improperly formatted line: {line}"
        version_part = parts[1]
        assert "." in version_part, f"Version part not properly formatted: {version_part}"
